468 payments_after_tds = sum([d.total_amount for d in frappe.get_list("Payment",filters={"invoice":self.name,"docstatus":1},fields=["total_amount"])]) 

469 payments_before_tds = sum([d.amount_received for d in frappe.get_list("Payment",filters={"invoice":self.name,"docstatus":1},fields=["amount_received"])]) 

470 settled_payments = sum([d.amount_received for d in frappe.get_list("Payment",filters={"utr_number":self.name,"docstatus":1},fields=["amount_received"])]) 

471 # received_payments = sum([d.received_amount for d in frappe.get_list("Vendor Recrun Mapping",filters={"vpo":self.name,"docstatus":1},fields=["received_amount"])]) 

472 received_payments = 0.0 

473 

474 # 10. Update Outstanding Amount, Total Payments & Final Amount  

475 self.grand_total = self.total_trip_cost + taxes 

476 self.rounded_total = frappe.utils.flt(self.grand_total,0) 

477 if self.is_debit_note != 1: 

478 # 10a. Update the Outstanding Amount & Final Amount 

479 if self.total_tax_and_charges and abs(self.total_tax_and_charges) > 0.0: 

480 self.net_amount = additions-deductions-advances-additional_payments - payments_before_tds + received_payments + settled_payments 

481 self.outstanding_amount = self.rounded_total - deductions - advances - additional_payments - payments_before_tds + received_payments + settled_payments 

482 else: 

483 self.net_amount = additions-deductions-advances-additional_payments - payments_after_tds + received_payments + settled_payments 

484 self.outstanding_amount = self.rounded_total - deductions - advances - additional_payments - payments_after_tds + received_payments + settled_payments 

485 

486 # 10b. Change the status of Recovered Trips 

487 if self.workflow_state in ["Vendor Recovery Pending","Settle on Future Trips"] and (abs(round(self.outstanding_amount)) == 0 or abs(round(self.outstanding_amount)) == 1): 

488 self.outstanding_amount = self.outstanding_amount 

489 apply_workflow(self, "Paid") 

490 else: 

491 self.net_amount = deductions 

492 self.outstanding_amount = deductions 

493 

494 self.rounding_adjustment = self.rounded_total - self.grand_total 

495 if abs(round(self.outstanding_amount)) == 1 or abs(round(self.outstanding_amount)) == 0: 

496 self.outstanding_amount = 0
