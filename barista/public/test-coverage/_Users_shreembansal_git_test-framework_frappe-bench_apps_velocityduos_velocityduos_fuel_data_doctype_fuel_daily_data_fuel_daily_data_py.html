@frappe.whitelist(allow_guest=True) 

def get_spacetime_fuel_average(cities, start_date, end_date): 

start_year, start_month, start_day = start_date.split("-") 

query_start_date = datetime( 

int(start_year), int(start_month), int(start_day)) 

 

end_year, end_month, end_day = end_date.split("-") 

query_end_date = datetime(int(end_year), int(end_month), int(end_day)) 

 

city_list = cities.split(",") 

data_points_considered = [] 

cumulative_petrol_price = 0.0 

cumulative_diesel_price = 0.0 

for city in city_list: 

try: 

fuel_price_data = frappe.get_doc('Fuel Daily Data', city) 

for fuel_price in fuel_price_data.price: 

current_year, current_month, current_day = fuel_price.date.split( 

"-") 

current_date = datetime(int(current_year), int( 

current_month), int(current_day)) 

if (current_date > query_start_date and current_date <= query_end_date) or current_date == query_end_date: 

data_points_considered.append(city + ":" + fuel_price.date) 

cumulative_diesel_price = cumulative_diesel_price + \ 

float(fuel_price.diesel_price) 

cumulative_petrol_price = cumulative_petrol_price + \ 

float(fuel_price.petrol_price) 

except Exception as e: 

print(e) 

pass 

if len(data_points_considered) == 0: 

petrol_average = 0 

diesel_average = 0 

else: 

petrol_average = cumulative_petrol_price / len(data_points_considered) 

diesel_average = cumulative_diesel_price / len(data_points_considered) 

return {"petrol_average_price": petrol_average, "diesel_average_price": diesel_average, "total_datapoints_considered": len(data_points_considered)}
