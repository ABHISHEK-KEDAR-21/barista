from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

class UserPreference(Document): 

def fetch_territory_details(self): 

territory = frappe.get_all("Territory",filters={'status':'Active'}) 

territory = [d.name for d in territory] 

territory.sort() 

roles = frappe.get_roles(self.user) 

if "Business Development" in roles: 

territory.insert(0,'All') 

 

return territory
