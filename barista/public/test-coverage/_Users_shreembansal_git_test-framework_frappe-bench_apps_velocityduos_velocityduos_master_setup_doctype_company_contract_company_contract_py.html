class CompanyContract(Document): 

def onload(self): 

self.set_onload('all_document_types', [dt.document_type for dt in frappe.db.get_all('Document Type', fields=[ 

'document_type'], filters={'disabled': 0, 'entity_type': 'Client'}, order_by="document_type")]) 

 

self.set_onload('insurance_document_types', [dt.document_type for dt in frappe.db.get_all('Document Type', fields=[ 

'document_type'], filters={'disabled': 0, 'entity_type': 'Insurance'}, order_by="document_type")]) 

 

def validate(self): 

if self.workflow_state == "Signed Contract Uploaded" and self.get_doc_before_save() and self.get_doc_before_save().workflow_state != self.workflow_state: 

if not self.ciaf_form or not self.ciaf_form.strip(): 

frappe.throw( 

"Please Attach the Customer & Invoice Appraisal Copy!") 

 

already_created_contracts = frappe.db.sql("""select name,workflow_state,start_date,end_date from `tabCompany Contract` where name!=%(name)s and company=%(company)s and ((%(start_date)s between start_date and end_date) or (%(end_date)s between start_date and end_date))""", { 

'company': self.company, 'start_date': self.start_date, 'end_date': self.end_date, 'name': self.name}, as_dict=1) 

 

if len(already_created_contracts): 

msg = '<table class="table"><tr><td>Company Contract</td><td> Workflow State </td><td>Start Date</td><td> End Date</td></tr>' 

for contract in already_created_contracts: 

msg = msg+'<tr><td>{0}</td><td> {1}</td><td> {2}</td><td>{3}</td></tr> '.format( 

contract['name'], contract['workflow_state'], contract['start_date'], contract['end_date']) 

frappe.throw('Company Contracts for this Company ({0}) already exist between these selected Start Date ({1}) and End Date ({2})<br>'.format( 

self.company, str(self.start_date), str(self.end_date))+msg+'</table>') 

 

if self.status == 'Active': 

contract_in_client = frappe.db.get_value( 

'Company', self.company, 'contract') 

if self.name != contract_in_client: 

frappe.throw( 

'This Contract is not currently set for the Client.<br>You are not allowed to change the Status to Active') 

 

def before_save(self): 

if self.workflow_state == "Signed Contract Uploaded": 

if not self.signed_contract_copy or not self.signed_contract_copy.strip(): 

frappe.throw("Please attach the Contract Physical Copy!") 

if not self.organisation_sign_off or not self.organisation_sign_off.strip(): 

frappe.throw( 

"Please enter the Organization Signed off Person!") 

if not self.company_sign_off or not self.company_sign_off.strip(): 

frappe.throw("Please enter the Company Signed Off Person!") 

# Check for Liability Details 

if self.insurance_type == "Shipper's risk": 

shortage_liability_amount = self.shortage_liability_amount if type( 

self.shortage_liability_amount) is float else float(self.shortage_liability_amount) 

damage_liability_amount = self.damage_liability_amount if type( 

self.damage_liability_amount) is float else float(self.damage_liability_amount) 

shortage_liability_percentage = self.shortage_liability_percentage if type( 

self.shortage_liability_percentage) is float else float(self.shortage_liability_percentage) 

damage_liability_percentage = self.damage_liability_percentage if type( 

self.damage_liability_percentage) is float else float(self.damage_liability_percentage) 

if self.liability_type == "Fixed amount per trip (Rs)" and (shortage_liability_amount <= 0 or damage_liability_amount <= 0): 

frappe.throw("Please provide the Liability Amount Details") 

elif self.liability_type == "% of invoiced value of the goods" and (shortage_liability_percentage <= 0 or damage_liability_percentage <= 0): 

frappe.throw("Please provide the Liability Percent Details") 

elif self.insurance_type == "Transporter's risk": 

if not self.hrr_type or not self.hrr_type.strip(): 

frappe.throw("Provide the HRR Details") 

else: 

if self.hrr_type == "% of Invoice Value" and self.hrr_percent <= 0: 

frappe.throw("Please provide the HRR Percentage") 

elif self.hrr_type == "Fixed Amount" and self.hrr_amount <= 0: 

frappe.throw("Please provide the HRR Amount") 

update_client_status(self) 

validate_penalty_details(self) 

validate_uom(self) 

 

def before_submit(self): 

update_client_status(self) 

validate_uom(self) 

 

def before_update_after_submit(self): 

update_client_status(self) 

validate_penalty_details(self) 

validate_uom(self) 

 

 

def update_client_status(self): 

status = self.status if self.workflow_state != "Unsigned Contract Created" else self.workflow_state 

frappe.db.sql("""update tabCompany set contract_status=%(status)s where contract=%(contract)s""", { 

"status": status, "contract": self.name}) 

 

 

def validate_penalty_details(self): 

if self.rate_type == "Agreed Rate" and (not self.vehicle_type_details or len(self.vehicle_type_details) == 0): 

frappe.throw( 

"Please provide the Vehicle Type wise Penalty Details in Non Placement Clause section") 

elif self.rate_type == "Market Hiring Rate" and self.vehicle_type_details: 

self.vehicle_type_details = None 

 

 

def validate_uom(self): 

uoms = frappe.get_all("Contract Uom", filters={}, 

fields=["name", "contract_type"]) 

contract_type = list(dict.fromkeys([d.contract_type for d in uoms])) 

uoms = {d.name: d.contract_type for d in uoms} 

row = 0 

for rates in self.contract_pricing: 

row += 1 

if rates.rate_type in contract_type: 

if not rates.unit_of_measure or not rates.unit_of_measure.strip(): 

frappe.throw( 

"Kindly Enter the Unit of Measure for Each Lane-wise Rates in Row #{0}".format(row)) 

elif rates.rate_type != uoms.get(rates.unit_of_measure): 

frappe.throw( 

"Rate Type and Unit of Measure does not match in Lane-wise Rates Row #{0}. Please Check".format(row)) 

 

 

def get_client_contract_data(lr_number): 

contract_data = frappe.db.sql("""SELECT * 

FROM `tabCompany Contract` cc 

INNER JOIN `tabCompany` c ON cc.name=c.contract 

INNER JOIN `tabConsignment` cmt ON cmt.customer=c.name 

INNER JOIN `tabLorry Receipt` lr ON lr.consignment=cmt.name 

WHERE lr.name=%(lorryReceiptNo)s and cc.status='Active'""", {"lorryReceiptNo": lr_number}, as_dict=True) 

 

return contract_data
