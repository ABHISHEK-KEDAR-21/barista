103 def update_shipping_price(self): 

104 freight_type = self.freight_type 

105 if freight_type == "Full Freight": 

106 if not self.freight_amount: 

107 frappe.throw("Please Enter Freight Amount!") 

108 else: 

109 self.shipping_price = self.freight_amount 

110 elif freight_type in ["Carton","Case","Weight"]: 

111 freight_per_unit = self.freight_per_unit if isinstance(self.freight_per_unit,float) else float(self.freight_per_unit) 

112 no_of_packages = 0.0 

113 weight = 0.0 

114 if len(self.items) > 0: 

115 for item in self.items: 

116 no_of_packages += (item.no_of_package if isinstance(item.no_of_package,float) else float(item.no_of_package)) 

117 weight += (item.weight if isinstance(item.weight,float) else float(item.weight)) 

118 if freight_type in ["Carton","Case"]: 

119 self.shipping_price = freight_per_unit * no_of_packages 

120 elif freight_type == "Weight": 

121 self.shipping_price = freight_per_unit * weight 

122 elif freight_type == "Distance": 

123 freight_per_unit = self.freight_per_unit if isinstance(self.freight_per_unit,float) else float(self.freight_per_unit) 

124 no_of_units = self.no_of_units if isinstance(self.no_of_units,float) else float(self.no_of_units) 

125 self.shipping_price = freight_per_unit * no_of_units 

126 

127 if self.shipping_price and self.shipping_price <= 0: 

128 frappe.throw("Shipping Price Cannot be less than or equal to 0")
