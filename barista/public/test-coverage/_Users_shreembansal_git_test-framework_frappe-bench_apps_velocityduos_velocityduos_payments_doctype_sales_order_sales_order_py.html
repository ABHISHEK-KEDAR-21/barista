from __future__ import unicode_literals 

import frappe 

from frappe.model.document import Document 

 

from velocityduos.api.erp_utils import sync_erp_doc 

 

class SalesOrder(Document): 

def validate(self): 

# 1. Validate LR Consignments 

if self.items and len(self.items)>0: 

for item in self.items: 

if item.consignment != self.consignment: 

frappe.throw("Consignment of LRs {0} is not {1}".format(item.lorry_receipt,self.consignment)) 

 

# 2. Validate Trip & Freight Type with Consignment 

consignment = frappe.get_doc("Consignment",self.consignment) 

if consignment.trip_type != self.trip_type: 

raise frappe.ValidationError("Trip Type do not match with Consignment") 

elif consignment.freight_type != self.freight_type: 

raise frappe.ValidationError("Freight Type do not match with Consignment") 

else: 

# 3. Validate Freight Price 

if consignment.freight_type != "Full Freight": 

# 3a. Validate Freight Per Unit 

cons_freight_per_unit = consignment.freight_per_unit if isinstance(consignment.freight_per_unit,float) else float(consignment.freight_per_unit) 

so_freight_per_unit = self.freight_per_unit if isinstance(self.freight_per_unit,float) else float(self.freight_per_unit) 

if cons_freight_per_unit != so_freight_per_unit: 

raise frappe.ValidationError("Freight Per Unit do not Match with Consignment") 

# 3b. Validate No of Units 

cons_units = consignment.no_of_units if isinstance(consignment.no_of_units,float) else float(consignment.no_of_units) 

so_units = self.no_of_units if isinstance(self.no_of_units,float) else float(self.no_of_units) 

if cons_units != so_units: 

if cons_units < so_units: 

raise frappe.ValidationError("No Per Unit do not Match with Consignment") 

lrs = frappe.get_all("Lorry Receipt",filters={"consignment":self.consignment},fields=["name"]) 

if len(lrs) == len(self.items): 

raise frappe.ValidationError("No Per Unit do not Match with Consignment") 

else: 

cons_shipping_price = consignment.shipping_price if isinstance(consignment.shipping_price,float) else float(consignment.shipping_price) 

so_shipping_price = self.shipping_price if isinstance(self.shipping_price,float) else float(self.shipping_price) 

if cons_shipping_price != so_shipping_price: 

if so_shipping_price > cons_shipping_price: 

raise frappe.ValidationError("Shipping Price do not Match with Consignment") 

lrs = frappe.get_all("Lorry Receipt",filters={"consignment":self.consignment},fields=["name"]) 

if len(lrs) == len(self.items): 

raise frappe.ValidationError("Shipping Price do not Match with Consignment") 

 

def before_submit(self): 

self.status = "Confirmed" 

 

def on_submit(self): 

sync_erp_doc(self.doctype,self.name,"Completed","Sales Order") 

 

def before_update_after_submit(self): 

if self.status != self.get_doc_before_save().status and self.status == "Completed": 

sync_erp_doc(self.doctype,self.name,"Completed","Delivery Note")
