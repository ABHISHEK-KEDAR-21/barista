import frappe 

import json 

import requests 

 

 

def get_coordinates(location): 

"""Returns Co-ordinates For Given Location""" 

locations = {'lat':0,'lng':0} 

api_config = frappe.get_single("Api Configuration") 

try: 

google_key="" 

if api_config.google_key and api_config.google_key.strip(): 

google_key = api_config.google_key.strip() 

else: 

frappe.throw("Google Api Key not Configured") 

 

geocode_url=f"https://maps.googleapis.com/maps/api/geocode/json?key={google_key}&address={location}" 

response=requests.get(geocode_url) 

if response.status_code==200: 

response=json.loads(response.content) 

if response['status']=="OK" and len(response['results'])>0 and len(response['results'][0]['geometry']) >0: 

locations = response['results'][0]['geometry']['location'] 

else: 

frappe.log_error(f"location - {location}\nresponse_code - {response.status_code}\nresponse - {response.content}", 'google_integration.get_coordinates') 

except Exception: 

frappe.log_error(f"location - {location}\nstack_trace - {frappe.get_traceback()}", 'google_integration.get_coordinates') 

frappe.throw(frappe.get_traceback()) 

 

return locations 

 

@frappe.whitelist(allow_guest=True) 

def get_distance(origin,destination): 

"""Returns Distance Between Two Places""" 

google_key="" 

distance = "0.0 km" 

try: 

if frappe.db.exists('Api Configuration','google_key'): 

api_config_doc=frappe.get_doc('Api Configuration','google_key') 

if api_config_doc: 

google_key=api_config_doc.value 

 

geocode_url="https://maps.googleapis.com/maps/api/distancematrix/json?key="+google_key+"&origins="+origin+"&destinations="+destination 

response=requests.get(geocode_url) 

if response.status_code==200: 

response=json.loads(response.content) 

if response['status']=="OK" and len(response['rows'])>0 and len(response['rows'][0]['elements']) >0: 

distance = response['rows'][0]['elements'][0]['distance']['text'] 

except Exception: 

frappe.log_error(f"location - {origin}-{destination}\nstack_trace - {frappe.get_traceback()}", 'google_integration.get_distance') 

frappe.throw(frappe.get_traceback()) 

return distance
